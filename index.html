CREATE TABLE discovered_files (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    file_path TEXT NOT NULL UNIQUE,
    file_name VARCHAR(255) NOT NULL,
    file_extension VARCHAR(20) NOT NULL,
    file_hash VARCHAR(64) NOT NULL, -- SHA-256 integrity confirmation
    department_id UUID REFERENCES departments(id) ON DELETE RESTRICT,
    owner_identity VARCHAR(255) NOT NULL,
    sensitivity_label VARCHAR(50) NOT NULL, -- Public, Internal, Confidential, Highly Sensitive
    computed_risk_score INT NOT NULL CHECK (computed_risk_score BETWEEN 0 AND 100),
    is_encrypted BOOLEAN DEFAULT FALSE,
    retention_expiry_date TIMESTAMP WITH TIME ZONE,
    last_scanned_timestamp TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

);

Core Authentication Execution Strategy 

# libs/core/auth/security_guard.py
import jwt
from fastapi import HTTPException, Security, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from pydantic import BaseModel
from typing import List

class TokenClaimsSchema(BaseModel):
    user_id: str
    email: str
    roles: List[str]
    organization_unit: str

class AccessRoleVerifier:
    def __init__(self, target_allowed_roles: List[str]):
        self.allowed_roles = target_allowed_roles
        self.signing_algorithm = "HS256"
        self.jwt_secret = "SYS_ENV_SECRET_DEEPLY_GUARDED_2026"

    async def __call__(self, token_header: HTTPAuthorizationCredentials = Security(HTTPBearer())) -> TokenClaimsSchema:
        raw_token = token_header.credentials
        try:
            payload = jwt.decode(raw_token, self.jwt_secret, algorithms=[self.signing_algorithm])
            token_claims = TokenClaimsSchema(**payload)
            
            # Match claims context tokens against strict RBAC definitions
            role_intersect = set(token_claims.roles).intersection(set(self.allowed_roles))
            if not role_intersect:
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail="RBAC Context Policy Denied: Insufficient authorization clearances."
                )
            return token_claims
        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token signature expired.")
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid crypt token structure.")

class AdvancedClassificationPipeline:
    def __init__(self):

        self.pan_compiled_regex = re.compile(r"[A-Z]{5}[0-9]{4}[A-Z]{1}")
        # Safeguarded storage handling enforcement for specific high-risk tokens
        self.aadhaar_compiled_regex = re.compile(r"[2-9]{1}[0-9]{3}\s[0-9]{4}\s[0-9]{4}")

    async def execute_content_extraction_trace(self, text_payload: str) -> Dict[str, Any]:
        detected_patterns = []
        
        # 1. Deterministic Regular Expression Verification Pass
        if self.pan_compiled_regex.search(text_payload):
            detected_patterns.append("Permanent Account Number (PAN)")
            
        if self.aadhaar_compiled_regex.search(text_payload):
            # Strict Redaction Compliance Enforcement Rule Activates Automatically
            detected_patterns.append("[Aadhaar Identity Number Redacted]")

        # 2. Semantic Structural Analysis Evaluation Mock Block via Local Agent Context
        # In a live architecture, this passes cleaned data to the Azure OpenAI or Ollama Client API

        computed_risk_weight = 45 if not detected_patterns else 85

        }
